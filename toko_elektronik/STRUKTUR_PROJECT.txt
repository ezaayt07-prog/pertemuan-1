toko_elektronik/
│
├── main.py                 # Program Utama (Entry Point)
│   └── Fungsi: Menjalankan aplikasi utama
│
├── models.py               # Data Models
│   ├── akun                # Dictionary akun pengguna
│   ├── barang              # Dictionary barang elektronik
│   └── total_penjualan     # Total penjualan toko
│
├── auth.py                 # Modul Autentikasi
│   ├── login()             # Fungsi login pengguna
│   └── register()          # Fungsi registrasi akun baru
│
├── inventory.py            # Modul Manajemen Inventori
│   ├── read_barang()       # Menampilkan daftar barang
│   ├── create_barang()     # Menambah barang baru
│   ├── update_barang()     # Mengubah data barang
│   ├── delete_barang()     # Menghapus barang
│   └── tampilkan_statistik() # Menampilkan statistik toko
│
├── transaction.py          # Modul Transaksi
│   ├── beli_barang()       # Proses pembelian barang
│   └── proses_beli()       # Interface pembelian untuk user
│
├── menu.py                 # Modul Menu Aplikasi
│   ├── menu_admin()        # Menu untuk admin
│   ├── menu_user()         # Menu untuk user
│   ├── tampilkan_menu_admin()    # Display menu admin
│   ├── tampilkan_menu_user()     # Display menu user
│   └── tampilkan_menu_utama()    # Display menu utama
│
├── utils.py                # Modul Utilitas
│   ├── clear()             # Membersihkan layar terminal
│   ├── hitung_total()      # Menghitung total harga
│   ├── format_rupiah()     # Format angka ke rupiah
│   ├── log_aktivitas()     # Mencatat aktivitas ke file
│   ├── tampilkan_log()     # Menampilkan log aktivitas
│   └── create_barang_table() # Membuat tabel barang (PrettyTable)
│
├── requirements.txt        # Dependencies (prettytable)
├── __init__.py            # Package initializer
└── README.md              # Dokumentasi lengkap
